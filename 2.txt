CREATE TABLE Sailors(sid NUMBER PRIMARY KEY,sname VARCHAR2(20),rating NUMBER CHECK(rating BETWEEN 1 AND 10),age NUMBER);

CREATE TABLE Boats(bid NUMBER PRIMARY KEY,bname VARCHAR2(20),color VARCHAR2(20));

CREATE TABLE Reserves(sid NUMBER,bid NUMBER,day DATE,PRIMARY KEY(sid,bid,day),FOREIGN KEY(sid) REFERENCES Sailors(sid),FOREIGN KEY(bid) REFERENCES Boats(bid));
