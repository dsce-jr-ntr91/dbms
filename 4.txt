CREATE TABLE BRANCH(BRANCH_NAME VARCHAR2(20) PRIMARY KEY,BRANCH_CITY VARCHAR2(10),ASSETS NUMBER);

CREATE TABLE ACCOUNT(ACCNO NUMBER(10) PRIMARY KEY,BRANCH_NAME VARCHAR2(20),BALANCE NUMBER(10),FOREIGN KEY(BRANCH_NAME) REFERENCES BRANCH(BRANCH_NAME));

CREATE TABLE CUSTOMER(CUSTOMER_NAME VARCHAR2(20) PRIMARY KEY,CUSTOMER_STREET VARCHAR2(20),CUST_CITY VARCHAR2(20));

CREATE TABLE DEPOSITOR(CUSTOMER_NAME VARCHAR2(20),ACCNO NUMBER(10),PRIMARY KEY(CUSTOMER_NAME,ACCNO),FOREIGN KEY(CUSTOMER_NAME) REFERENCES CUSTOMER(CUSTOMER_NAME),FOREIGN KEY(ACCNO) REFERENCES ACCOUNT(ACCNO) ON DELETE CASCADE);

CREATE TABLE LOAN(LOAN_NO NUMBER PRIMARY KEY,BRANCH_NAME VARCHAR2(20),AMOUNT NUMBER,FOREIGN KEY(BRANCH_NAME) REFERENCES BRANCH(BRANCH_NAME));

CREATE TABLE BORROWER(CUSTOMER_NAME VARCHAR2(20),LOAN_NO NUMBER,FOREIGN KEY(CUSTOMER_NAME) REFERENCES CUSTOMER(CUSTOMER_NAME),FOREIGN KEY(LOAN_NO) REFERENCES LOAN(LOAN_NO));
